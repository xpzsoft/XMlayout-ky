    $(document).ready(function(){
        var xm = $.xmlayout.layout();

        var data = xm.design(
          3,//rows
          2,//cols
          1,//按照列划分
          [//矩阵cols × (rows + 1) = 2 × 4
            [1,1,1,1],
            [2,1,2,2]
          ],
          [//矩阵，取消第1行第1列子容器的右键菜单和工具条
            ["1:1", false, false]
          ]
        );

        //对data中的第2行第2列的子容器再次划分
        var data1 = xm.design(2,1,0,null,null,xm.getDesignNode("2:2", data));

        //对data1中的第2行第1列的子容器再次划分
        xm.design(2,2,1,null,null,xm.getDesignNode("2:1", data1));

        xm.init({
          data : data,
          panel : $("#main"),
          drag_bar_unit : 5,
          isborder : true,
          timer : 2000
        });

        xm.getPanels("2:2;2:1;1:1").css("background-color","red");
      });
